```python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        elements = {}
        for i, num in enumerate(nums):
            rest_num = target - num
            if rest_num in elements:
                return [i, elements[rest_num]]
            elements[num] = i
```
